/* Многострочный комментарий (комментарии не занимают память) */
#include - подключить файл (библиотеку)

void setup() {} - всё находящееся внутри {} будет выполнено 1 раз при загрузке Ардуино
void loop() {} - всё находящееся внутри {} бесконечно повторяется сверху вниз

### Типы данных: #############################################

boolean myboolean = 0 или 1;
char mychar = от -128 до 127;
byte mybyte = от 0 до 255;
int myint = от -32768 до 32 767; 
unsignet int = 0...65535;
word = 0...65535;
long = - 2147483648...2147483647;
float = 10.3'

### Работа с последовательным портом: ##########################

Serial - объект библиотеки для работы с последовательным портом
Serial.begin(9600);	открыть порт на 9600бод
Serial.print(); 	вывести в порт, переменные и цифры напрямую, а текст в кавычках ""
Serial.println();	вывод с переводом строки;
Serial.println(val, n);	вывод переменной float c n знаков после запятой.
Serial.read();		прочитать данные с порта.

### Условный оператор: ##########################################

if() {}		проверяет условие () и выполняет код в {} если оно верно
else{}		если неверно, выполняет эту часть кода
else if() {}	если неверно, то проверяет новое
a==b	если a равно b.
a!=b	если a не равно b.
> 	если больне.
<	если меньше.
>=	если больше или равно.
<=	если меньше или равно.

&&	логическое И - одно условие и второе.
||	логическое ИЛИ - одно либо второе.
!	отрицание.

switch (val) { рассматриваем переменную val
case 1: - если она равно 1, выполнить код тут.
break;
case 2: - если она равно 2, выполнить код тут.
break;
default: - если что-то еще, выполнить код тут.
}

### Временные функции: ##############################################

delay()		задержка в миллисекундах
delayMicroseconds()	задержка в микросекундах.
millis()	возвращает количестко миллисекунд прошедших с момента включения мк.
micros()	то же самое в микросекундах.

### Цифровые порты: #################################################

pinMode(pin,mode);	настроить порт.
pin	номер порта.
mode	режим его работы.
	INPUT - вход, принимает сигнал.
	OUTPUT - выход, отдает сигнал.
	INPUT_PULLUP - вход с подтяжкой к 5В.

### Генерация цифрового сигнала: #####################################

digitalWrite(pin,signal);	подать цифровой сигнал.
signal	какой сигнал подаем.
	LOW - 0, или false - 0В.
	HIGH -1, или true - 5В.
digitalRead(pin);	чтение цифрового сигнала.
!digitalRead(pin); 	инвертировать сигнал.
